function calculateOptimalKit(totalBudget, activeCategoryNames, selectedProfileId) { let itemsList = []; const profile = profiles[selectedProfileId]; // 1. Normalize weights for only the categories the user actually selected let totalWeight = 0; activeCategoryNames.forEach(cat => totalWeight += (profile[cat] || 0)); let targetBudgets = {}; activeCategoryNames.forEach(cat => { targetBudgets[cat] = ((profile[cat] || 0) / totalWeight) * totalBudget; const items = productDatabase[cat] || []; items.forEach(item => { itemsList.push({ catName: cat, baseItem: item, currentTierIdx: 0 }); }); }); let hadToOverride = itemsList.reduce((sum, i) => sum + i.baseItem.tiers[0].price, 0) > totalBudget; // PHASE 1: Targeted Category Upgrades let canUpgradePhase1 = true; while(canUpgradePhase1) { canUpgradePhase1 = false; activeCategoryNames.forEach(cat => { let catItems = itemsList.filter(i => i.catName === cat); let currentCatTotal = catItems.reduce((sum, i) => sum + i.baseItem.tiers[i.currentTierIdx].price, 0); let candidates = catItems.filter(i => { if (i.currentTierIdx >= i.baseItem.tiers.length - 1) return false; let costToUpgrade = i.baseItem.tiers[i.currentTierIdx + 1].price - i.baseItem.tiers[i.currentTierIdx].price; return (currentCatTotal + costToUpgrade) <= targetBudgets[cat]; }); if (candidates.length > 0) { // Sort by cheapest upgrade within this category candidates.sort((a,b) => (a.baseItem.tiers[a.currentTierIdx + 1].price - a.baseItem.tiers[a.currentTierIdx].price) - (b.baseItem.tiers[b.currentTierIdx + 1].price - b.baseItem.tiers[b.currentTierIdx].price)); candidates[0].currentTierIdx++; canUpgradePhase1 = true; } }); } // PHASE 2: Global Sweep for Leftover Budget let currentGlobalTotal = itemsList.reduce((sum, i) => sum + i.baseItem.tiers[i.currentTierIdx].price, 0); let canUpgradePhase2 = true; while (canUpgradePhase2 && currentGlobalTotal < totalBudget) { let candidates = itemsList.filter(i => { if (i.currentTierIdx >= i.baseItem.tiers.length - 1) return false; let costToUpgrade = i.baseItem.tiers[i.currentTierIdx + 1].price - i.baseItem.tiers[i.currentTierIdx].price; return (currentGlobalTotal + costToUpgrade) <= totalBudget; }); if (candidates.length === 0) { canUpgradePhase2 = false; } else { // Prioritize upgrading items from their most valued categories first candidates.sort((a, b) => { let weightA = (profile[a.catName] || 0) / totalWeight; let weightB = (profile[b.catName] || 0) / totalWeight; if (weightB !== weightA) return weightB - weightA; // Tie-breaker: cheapest upgrade let costA = a.baseItem.tiers[a.currentTierIdx + 1].price - a.baseItem.tiers[a.currentTierIdx].price; let costB = b.baseItem.tiers[b.currentTierIdx + 1].price - b.baseItem.tiers[b.currentTierIdx].price; return costA - costB; }); let costToUpgrade = candidates[0].baseItem.tiers[candidates[0].currentTierIdx + 1].price - candidates[0].baseItem.tiers[candidates[0].currentTierIdx].price; candidates[0].currentTierIdx++; currentGlobalTotal += costToUpgrade; } } // Package the final data let kitByCategory = {}; let catTotals = {}; activeCategoryNames.forEach(cat => { kitByCategory[cat] = []; catTotals[cat] = 0; }); itemsList.forEach(i => { let selectedProduct = i.baseItem.tiers[i.currentTierIdx]; kitByCategory[i.catName].push(selectedProduct); catTotals[i.catName] += selectedProduct.price; }); return { kitByCategory, catTotals, grandTotal: currentGlobalTotal, hadToOverride, targetBudgets }; }
